  <script>
    function updateImageInputs() {
      const container = document.getElementById("image-uploads");
      container.innerHTML = "";
      const selected = document.querySelector("input[name='csomag']:checked").value;
      let count = 0;
      if (selected === "Alap") count = 2;
      if (selected === "Prémium") count = 3;
      if (selected === "Extra") count = 5;
      for (let i = 1; i <= count; i++) {
        const input = document.createElement("input");
        input.type = "file";
        input.accept = "image/*";
        input.id = "image" + i;
        input.name = "image" + i;
        container.appendChild(input);
      }
    }
    document.querySelectorAll("input[name='csomag']").forEach(radio => {
      radio.addEventListener("change", updateImageInputs);
    });
    updateImageInputs();

    // Demo: csak prémium/extránál jelenjen meg a fizetési terület!
    document.querySelectorAll("input[name='csomag']").forEach(radio => {
      radio.addEventListener("change", () => {
        const payArea = document.getElementById("pay-area");
        const selected = document.querySelector("input[name='csomag']:checked").value;
        payArea.innerHTML = "";
        if (selected === "Prémium" || selected === "Extra") {
          payArea.innerHTML = "<b>PayPal/kártyás fizetés itt lesz (helykitöltő)</b>";
        }
      });
    });

    // Ha az oldalt fejleszted, ide jönne a valódi PayPal JS/Stripe fizetési integráció
  </script>
